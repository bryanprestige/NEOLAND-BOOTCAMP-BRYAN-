    document.addEventListener('DOMContentLoaded', onDomContentLoaded)

function onDomContentLoaded () {
        
        const defaultHeader = document.getElementById('dance-header') 
        const bachatatHeader = document.getElementById('bachata-header') 
        const salsaHeader = document.getElementById('salsa-header') 
        const zoukHeader = document.getElementById('zouk-header') 

        defaultHeader.addEventListener('onmouseover',dancesDisplay) 
        bachatatHeader.addEventListener('click',clickOnBachataPage)
        salsatHeader.addEventListener('click',clickOnSalsaPage)
        zouktHeader.addEventListener('click',clickOnZoukPage)

        updateDefaultFeed()
    }

function danceDisplay (e) {
      displayDancesWindow ()
}
function displayDancesWindow () {
    
     if (clickOnBachataPage(e)){
        goToBachataPage(){
            const defaultHeader = document.getElementById('dance-header') 
            const bachatatHeader = document.getElementById('bachata-header') 
            const salsaHeader = document.getElementById('salsa-header') 
            const zoukHeader = document.getElementById('zouk-header') 

            defaultHeader.style.display = 'none'
            bachatatHeader.style.display = 'block'
            salsaHeader.style.display = 'none'
            zouktHeader.style.display = 'none'

            updateBachatatFeed()
        }
     } else if (clickOnSalsaPage(e)){
        goToSalsaPage(){
            const defaultHeader = document.getElementById('dance-header') 
            const bachatatHeader = document.getElementById('bachata-header') 
            const salsaHeader = document.getElementById('salsa-header') 
            const zoukHeader = document.getElementById('zouk-header') 

            defaultHeader.style.display = 'none'
            bachatatHeader.style.display = 'none'
            salsaHeader.style.display = 'block'
            zouktHeader.style.display = 'none'

            updateSalsaFeed()
        }
     } else if (clickOnZoukPage(e)){
        goToZoukPage() {
            const defaultHeader = document.getElementById('dance-header') 
            const bachatatHeader = document.getElementById('bachata-header') 
            const salsaHeader = document.getElementById('salsa-header') 
            const zoukHeader = document.getElementById('zouk-header') 

            defaultHeader.style.display = 'none'
            bachatatHeader.style.display = 'none'
            salsaHeader.style.display = 'none'
            zouktHeader.style.display = 'block'

            updateZoukFeed()
        }
     }
}

function addEventToList{    
    let EVENT = document.createElement('li')
    let olEVENTS = document.getElementById('lista-events')
    EVENT.class = 'event-style'
    EVENT.innerText = 'Soy un nuevo evento'

    EVENTS.push(EVENT)
    EVENTS.appenchild (EVENT)
}
